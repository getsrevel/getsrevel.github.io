Web Scraping

Scraper eksempel

Dette er et eksempel på en scraper der finder alle links på en side, og skriver deres URL ud i konsollen.

const cheerio = require("cheerio");
const axios = require("axios");

// Swap this for the url you want to scrape
const url =
  "https://en.wikipedia.org/wiki/List_of_Presidents_of_the_United_States";

axios
  .get(url)
  .then(response => {
    let $ = cheerio.load(response.data);
    $("a").each(function(i, e) {
      let links = $(e).attr("href");
      console.log(links);
    });
  })
  .catch(function(e) {
    console.log(e);
  });

For at køre eksemplet skal der installeres nogle biblioteker fra npm.

Først skal der lavet et projekt så node kan finde ud af at køre din fil, og har en package.json fil til at holde styr på projektet og afhængigheder af biblioteksmoduler.

npm init -y

Dernæst har du mulighed for at installere disse afhængigheder.

npm --save install axios
npm --save install cheerio

Nu kan du køre eksemplet med denne kommando:

node simple-scraper.js

Her er et eksempel med samme funktionalitet, men hvor der bruges async/await syntaks i stedet for promise.then().

// import the library modules you installed from npm
const cheerio = require("cheerio");
const axios = require("axios");

// Define the scraping algorithm
async function runScraper(url) {
  try {
    const html = (await axios.get(url)).data;
    const $ = cheerio.load(html);

    $("a").each(function(i, e) {
      const url = $(e).attr("href");
      console.log(url);
    });
  } catch (error) {
    console.log("error: " + error);
  }
}

// Swap this for the url you want to scrape
const url =
  "https://en.wikipedia.org/wiki/List_of_Presidents_of_the_United_States";

// run the scraper by calling the function
runScraper(url);
